[main]
log=путь к логу
pid=путь к pid-файлу
sid=перечисление sid, имя1, имя2
host=имя собственного хоста, по-дефолту не указывается. Если нет - определять самостоятельно.
ip=ip собственного хоста, по дефолту не указывается. Здесь живёт пассивный агент.
port=порт собственного хоста, по дефолту не указывается
delay=пауза между отправками на сервер. Дефолт - 60 сек
server=имя zabbix-сервера. Можно указывать с номером порта

[имя1]
login=логин
password=пароль
url=jdbc url
driver=название jdbc-драйвера
sql=каталог с запросами. Запросы в этом каталоге исполняются для данного tns/sid
delay=индивидуальный забор данных. Дефолт - 60 сек.
timeout=время ожидания ответа запроса. Можно не указывать, по-умолчанию - 60 сек.
type=text/num/float. Дефолт - text
